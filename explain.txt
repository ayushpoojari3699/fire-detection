def send_email(subject, body, image_path):
    msg = MIMEMultipart()
    msg['From'] = formataddr(('Fire Detection System', sender_email))
    msg['To'] = receiver_email
    msg['Subject'] = subject

    msg.attach(MIMEText(body, 'plain'))

    with open(image_path, 'rb') as img_file:
        img = MIMEImage(img_file.read())
        img.add_header('Content-Disposition', 'attachment', filename=image_path)
        msg.attach(img)

    try:
        with smtplib.SMTP('smtp.gmail.com', 587) as server:
            server.starttls()
            server.login(sender_email, password)
            server.sendmail(sender_email, receiver_email, msg.as_string())
            print("Email sent successfully.")
    except Exception as e:
        print(f"Failed to send email: {e}")
send_email: Function to compose and send an email.
MIMEMultipart: Creates a multipart email.
MIMEText: Adds text content to the email.
MIMEImage: Attaches the image to the email.

6. Frame Processing

def run_inference(frame):
    img = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    img = cv2.resize(img, (640, 640))  # Resize for faster processing
    img = np.transpose(img, (2, 0, 1))  # HWC to CHW
    img = np.expand_dims(img, axis=0)
    img = np.ascontiguousarray(img)
    img = torch.from_numpy(img).to(device)
    img = img.half() if model.fp16 else img.float()  # uint8 to fp16/32
    img /= 255.0  # 0 - 255 to 0.0 - 1.0
    pred = model(img)
    pred = non_max_suppression(pred, 0.25, 0.45, agnostic=False)
    return pred
run_inference: Function to preprocess the frame, run it through the YOLOv5 model, and apply non-max suppression.
cv2.cvtColor: Converts the frame from BGR to RGB.
cv2.resize: Resizes the frame for faster processing.
np.transpose: Reorders dimensions for the model input.
torch.from_numpy: Converts the numpy array to a PyTorch tensor.
model(img): Runs inference on the image.
non_max_suppression: Removes duplicate detections.



7. Frame Processing for Fire Detection

def process_frame(frame):
    pred = run_inference(frame)
    fire_detected = False
    for i, det in enumerate(pred):  # detections per image
        annotator = Annotator(frame, line_width=2, example=str(names))
        if len(det):
            for *xyxy, conf, cls in reversed(det):
                if names[int(cls)] == 'fire':  # Assuming 'fire' is the class name for fire
                    fire_detected = True
                # Draw bounding box and label
                label = f'{names[int(cls)]} {conf:.2f}'
                annotator.box_label([int(xyxy[0]), int(xyxy[1]), int(xyxy[2]), int(xyxy[3])], label, color=colors(cls))
        frame = annotator.result()

    return frame, fire_detected
process_frame: Function to run inference and annotate the frame.
annotator.box_label: Draws bounding boxes and labels on the frame.
fire_detected: Boolean indicating if fire is detected.


8. Capture and Process Frames

def capture_and_process():
    while True:
        ret, frame = cap.read()
        if not ret:
            break

        frame, fire_detected = process_frame(frame)

        if fire_detected:
            timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
            image_path = f'detected_fire_{timestamp}.jpg'
            cv2.imwrite(image_path, frame)
            send_email(
                subject="Fire Detected!",
                body="Fire has been detected. Please find the attached image.",
                image_path=image_path
            )

        cv2.imshow('Fire Detection', frame)

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
capture_and_process: Function to continuously capture frames from the webcam, process them, and send an email if fire is detected.
cv2.imwrite: Saves the detected frame with a timestamp.
send_email: Sends an email with the attached image.
cv2.imshow: Displays the frame with annotations.
